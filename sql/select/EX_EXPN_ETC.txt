      ,SEQ
      ,OCCR_ACC_CD
      ,REV_ACC_CD
      ,BSS_YMD
      ,END_YMD
      ,OPEN_YMD
      ,VAT_CD
      ,CST_APPV_METH
      ,OCCR_YMD
      ,ECAL_AMT
      ,SUPPLY_AMT
      ,TAX_AMT
      ,DTLS
      ,CUST_NM
      ,CUST_NO
      ,CUST_CD
      ,CHRPE_NM
      ,PRAMT_SLIP_NO
      ,STTL_AMT
      ,TOT_RTN_AMT
      ,ELEC_APPV_STAT_CD
      --,FVNSMNG.FC_CM_DECRYPT_AES(ELEC_APPV_DOC_NO)
      ,'NULL' AS ELEC_APPV_DOC_NO
      ,ECAL_STAT_CD
      ,ELEC_APPV_YMD
      ,SLIP_HANDL_YN
      ,SLIP_NO
      ,RGSPE_ID
      ,RGS_DH
      ,UTIL_YN
      ,KNDNR_NM
      ,KUNDE_PA
      ,COM_CD
      ,YY
      ,(SELECT DPT.CCTR
          FROM FMSMNG.EX_EXPN@PRDSEL EEP
             , HRM.EMPLOYEE@PRDSEL EPL
             , HRM.DEPARTMENT@PRDSEL DPT
         WHERE EEP.ECAL_NO = ECA.ECAL_NO
           AND EEP.WRT_EMPNO = EPL.EMP_CD
           AND EPL.DEPT_CD = DPT.DEPT_CD) AS CST_CENTR_CD
  FROM FMSMNG.EX_EXPN_ETC@PRDSEL ECA
 WHERE COM_CD = '1000'
   AND OCCR_YMD BETWEEN TO_CHAR(ADD_MONTHS(SYSDATE, -1), 'YYYYMM')||'01' AND TO_CHAR(LAST_DAY(ADD_MONTHS(SYSDATE, -1)), 'YYYYMMDD')
